=LET(
    games, B2,
    wins, C2,
    losses, D2,
    kills, E2,
    deaths, F2,
    assists, G2,
    winrate, wins/games,
    kill_involve, L2,
    
    /* Adjusted efficiency: assists weighted higher (supports get credit), deaths softened */
    efficiency, IF(deaths=0, 8, (kills + assists*1.35)/(deaths + 3))^0.85,
    
    /* Main base score */
    base, 
        (winrate * 520) + 
        (kill_involve * 340) + 
        (efficiency * 28) + 
        (games * 5.5) - 
        (losses * 4.2) - 
        ((1 - winrate)^1.12 * 265) - 
        IF(wins=0, 310, 0),
    
    /* Volume multiplier (same philosophy as yours - diminishing returns) */
    vol_mult, MIN(LOG(games + 3.5), 1.15),
    
    /* Role-neutral bonuses (same style as current) */
    bonuses, 
        IF(AND(winrate=1, games>=3), 165, 0) +
        IF(AND(winrate>=0.75, winrate<1, games>=3), 105, 0) +
        IF(AND(winrate>=0.6, winrate<0.75, games>=3), 65, 0) +
        IF(efficiency > 3.8, 28, 0) +
        IF(efficiency > 6, 14, 0),
    
    /* Low-game safeguard (same as current) */
    IF(games < 3,
        MIN(base * 0.85, 520) + bonuses,
        base * vol_mult + bonuses
    )
)
